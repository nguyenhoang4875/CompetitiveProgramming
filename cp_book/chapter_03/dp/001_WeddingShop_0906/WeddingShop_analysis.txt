    memo[C + 1][M + 1]; // memo[i][j] số tiền tối đa mua được i-th first items và dùng ngân sách j 
    memset(memo, -1, sizeof(memo));

    function<int(int, int)> f = [&](int idx, int b) {
        if (b < 0) return (int)-1e9;
        if (idx == C) return M - b;

        int &ans = memo[idx][b];
        if (ans != -1) return ans;

        for (int k = 1; k <= price[idx][0]; ++k) {
            ans = max(ans, f(idx + 1, b - price[idx][k]));
        }
        return ans;
    };
